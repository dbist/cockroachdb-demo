Step 1. Initialize the workload (on a single client)

docker run --rm -it -m 15g  --cpus=4 \
 postgres pgbench \
    --initialize \
    --host=${PGHOST} \
    --username=${PGUSER} \
    --port=${PGPORT} \
    --no-vacuum \
    --scale=${SCALE} \
    --foreign-keys \
    ${PGDATABASE}

Step 2. Run the workload (on each client)

docker run --rm -it -m 15g --cpus=4 \
    --volume="$(pwd)"/tpcb-cockroach.sql:/home/ubuntu/tpcb-cockroach.sql \
    postgres pgbench \
    --host=${PGHOST} \
    --username=${PGUSER} \
    --port=${PGPORT} \
    --no-vacuum \
    --file=/home/ubuntu/tpcb-cockroach.sql@1 \
    --client=30 \
    --jobs=30 \
    --scale=${SCALE} \
    --time=6000 \
    --failures-detailed \
    --max-tries=10 \
    --protocol=prepared \
    -P 5 \
    ${PGDATABASE}

docker run --rm -it -m 15g --cpus=4 \
    postgres pgbench \
        --host=${PGHOST} \
        --username=${PGUSER} \
        --port=${PGPORT} \
        --no-vacuum \
        --builtin=tpcb-like@1 \
        --client=30 \
        --jobs=30 \
        --scale=${SCALE} \
        --time=1800 \
        --failures-detailed \
        --max-tries=10 \
        --protocol=prepared \
        -P 5 \
        --connect \
        ${PGDATABASE}
